class Solution {
    public int numJewelsInStones(String J, String S) {
        Set<Character> jewelSet = new HashSet<Character>();
        for(int i = 0; i < J.length(); i++) {
            jewelSet.add(J.charAt(i));
        }
        int count = 0;
        for(int i = 0; i < S.length(); i++) {
            if(jewelSet.contains(S.charAt(i))) {
                count++;
            }
        }
        return count;
    }
}
